### 2.3 已有代码的具体问题定位

#### 问题A：多片段投票已实现但偏弱
- **位置**: `processor.py:910-957` `_identify_song_from_multiple_clips`
- **现状**: 固定取 20%/50%/80% 三段，各截 ~8-20s
- **缺陷**:
  1. 片段长度固定，不适应长短差异大的歌曲
  2. 投票阈值要求 ≥2/3 通过才采纳，但只有3段 → 实际要求全一致
  3. 无"部分匹配加分"机制（如 track_id 不完全一致但属于同一歌手的不同歌）
  4. 底层 `identify_song_from_file()` 返回值结构不透明

#### 问题B：OCR 过滤规则太基础
- **位置**: `processor.py:649-664` `_is_likely_song_title_text`
- **现有规则**:
  - 长度 2-40 字符
  - 含 credits 关键词（作词/作曲等）→ 过滤
  - 含连续标点符号句子 → 过滤
  - 含噪声词（谢谢/大家/掌声等）→ 过滤
- **缺少**:
  - ❌ 歌词句子过滤（如 "我爱你"、"孤单北半球" 被当标题）
  - ❌ 乱码/装饰符号检测
  - ❌ 中文字数 vs 英文字数的合理性判断
  - ❌ OCR 置信度权重（当前只看文字不看分数）

#### 问题C：误拆回并阈值过严
- **位置**: `processor.py:1019-1039` `_should_merge_adjacent_songs`
- **现有4条件**:
  - (a) OCR标准化标题完全相同
  - (b) track_id 相同
  - (c) 标题相似 > 0.85 **且** 边界置信 < 0.65
  - (d) 双未知 + 低置信 + 间隔 ≤ 8s
- **缺陷**:
  - 条件(c) 要求 0.85 相似度太高，演唱会 OCR 错别字后很难达到
  - 合并后无"最大歌曲时长"自适应检查
  - 缺少"音频特征相似性"作为第5个合并信号

#### 问题D：AST 场景分类完全缺失
- **文件**: `classifier.py` 整个文件 (~266行)
- **功能**: Audio Spectrogram Transformer 做 scene 分类（speech/clap/crowd/music）
- **状态**: 写好了但 processor.py **从未 import 或调用**
- **后果**: 讲话串场、观众合唱、鼓掌欢呼全部无法识别，被错误归类为 verse/chorus

#### 问题E：无时序平滑
- **现状**: SongFormer 输出原始帧级标签，直接使用
- **问题**: 正常歌曲结构是 A-A-B-A-B-C-D（verse-chorus-verse-chorus-bridge-outro），但现场版可能有噪声帧导致中间插入异常标签
- **参考**: Springer 2026 综述推荐 Viterbi/DP 动态规划做时序平滑

```python
def _identify_song_from_multiple_clips_v2(self, song, y_full, sr):
    """v2: 自适应多片段识曲投票"""
    dur = song.duration
    
    # 根据歌曲长度动态决定段数和位置
    if dur < 60:
        positions = [0.2, 0.5, 0.8]
        clip_len = min(12.0, max(8.0, dur * 0.3))
    elif dur < 120:
        positions = [0.15, 0.3, 0.5, 0.7, 0.85]
        clip_len = min(15.0, max(8.0, dur * 0.25))
    else:
        positions = [0.1, 0.2, 0.35, 0.5, 0.65, 0.8, 0.9]
        clip_len = min(18.0, max(10.0, dur * 0.2))
    
    # 避开首尾可能的静默/过渡段
    safe_start = min(5.0, dur * 0.08)
    safe_end = max(dur - 5.0, dur * 0.92)
    
    matches = []
    for pos in positions:
        center = song.start_time + dur * pos
        center = max(song.start_time + safe_start, min(song.start_time + safe_end, center))
        
        c_start = max(song.start_time, center - clip_len / 2)
        c_end = min(song.end_time, center + clip_len / 2)
        c_start = max(song.start_time, c_end - clip_len)
        
        y_clip = y_full[int(c_start * sr):int(c_end * sr)]
        match = self._identify_song_from_audio_clip(y_clip, sr)
        if match:
            matches.append(match)
    
    if not matches:
        return None
    if len(matches) == 1:
        return matches[0]
    
    # 柔和投票：track_id一致=3分，title相似>0.8=2分，同artist=1分
    from collections import defaultdict
    score_map = defaultdict(float)
    cand_map = {}
    for m in matches:
        tid = getattr(m, 'track_id', '') or ''
        title = getattr(m, 'title', '') or ''
        artist = getattr(m, 'artist', '') or ''
        key = tid if tid else f"na::{self._normalize_title_for_compare(title)}"
        
        base_score = 1.0
        if tid:
            score_map[key] += 3.0
        else:
            best_sim = max(
                (SequenceMatcher(None, self._normalize_title_for_compare(title),
                              self._normalize_title_for_compare(getattr(m2, 'title', ''))).ratio()
                for m2 in matches if m is not m2
            ) or [0]
            if best_sim > 0.8:
                score_map[key] += 2.0
            score_map[key] += 1.0
        
        cand_map[key] = m
    
    best_key = max(score_map, key=score_map.get)
    vote_score = score_map[best_key]
    
    n_matches = len(matches)
    if vote_score >= 3.0 and n_matches >= 2:
        return cand_map[best_key]
    elif vote_score >= 2.0 and n_matches >= 3:
        return cand_map[best_key]
    elif vote_score >= 1.0 and n_matches >= 1:
        return cand_map[best_key]
    # 全部丢弃，走 OCR 路径
```

#### 优化 A2：强化 OCR 文字质量过滤

具体改动 (`processor.py`):

```python
@staticmethod
def _ocr_title_quality_score(text: str, ocr_confidence: float = 0.5) -> dict:
    """对 OCR 结果做多维评分"""
    score = 0.0
    reasons = []
    
    # 1. 长度得分 (0~30分)
    clen = len(text)
    if 2 <= clen <= 8:
        score += 25; reasons.append("len_ok")
    elif 9 <= clen <= 15:
        score += 18; reasons.append("len_acceptable")
    elif 16 <= clen <= 25:
        score += 10; reasons.append("len_long")
    else:
        reasons.append(f"len_bad({clen})")
    
    # 2. 中文占比 (0~20分)
    cn_chars = len(re.findall(r'[\u4e00-\u9fff]', text))
    ratio = cn_chars / max(len(text), 1)
    if 0.3 <= ratio <= 0.95:
        score += 20; reasons.append("cn_ratio_good")
    else:
        reasons.append(f"cn_ratio_bad({ratio:.2f})")
    
    # 3. 歌词模式扣分（关键新增！）
    lyric_patterns = [
        (r'[。？！?!…]+', -15, "有句末标点"),
        (r'^.{0,3}(我爱|你是|这是|我想|不要|可以|让我)', -12, "疑似歌词"),
        (r'(喜欢|恨|心|爱|哭|笑|痛|梦|飞|走|留)', -10, "常见动词"),
        (r'(呢|吧|吗|呀|啦|噢|喔)', -8, "语气词过多"),
        (r'^(我们|他们|你们|大家|所有)+$', -10, "代词开头"),
    ]
    for pat, penalty, reason in lyric_patterns:
        if re.search(pat, text):
            score += penalty
            reasons.append(reason)
    
    # 4. 结构特征加分
    if re.match(r'^[\u4e00-\u9fff\w\s]{2,8}$', text):
        score += 10; reasons.append("clean_short")
    if re.search(r'[A-Za-z]{2,}', text) and cn_chars >= 1:
        score += 5; reasons.append("mixed_cn_en")
    
    # 5. Credits 行硬过滤
    credit_kw = ['作词', '作曲', '编曲', '制作', '演唱', 'lyrics', 'arranged']
    if any(kw in text for kw in credit_kw):
        return {'score': score, 'is_title': False, 'reasons': ['credits_line']}
    
    # 6. 乱码检测
    garbage = sum(1 for ch in text if ord(ch) > 0x3000 or ch in '◎☆●○※□■△▽▐|')
    if garbage > len(text) * 0.2:
        score -= 30; reasons.append(f"garbage({garbage}chars)")
    
    is_title = (score >= 25)
    return {'score': round(score, 1), 'is_title': is_title,
            'confidence': min(score / 50.0, 1.0), 'reasons': reasons}
```

#### 优化 B1：放松误拆回并条件

目标: 降低"一首拆多首"的发生率

具体改动 (`processor.py` `_should_merge_adjacent_songs`):

```python
def _should_merge_adjacent_songs_v2(self, left, right):
    """v2: 放松合并条件，增加音频特征信号"""
    gap = right.start_time - left.end_time
    
    # === 原有条件（保留，略微放宽）===
    
    # (a) OCR 标题完全相同
    if left._ocr_title_norm and right._ocr_title_norm:
        if left._ocr_title_norm == right._ocr_title_norm:
            return True, "ocr_exact"
    
    # (b) track_id 相同
    if left.track_id and right.track_id and left.track_id == right.track_id:
        return True, f"same_track({left.track_id})"
    
    # (c) 标题相似度：从 0.85 降到 0.70，去掉边界置信度硬约束
    sim = self._title_similarity(left.song_title, right.song_title)
    if sim > 0.70:
        return True, f"title_sim({sim:.2f})"
    
    # (d) 双未知 + 低置信：间隔从 8s 放宽到 15s
    if (self._is_unknown_title(left.song_title) and 
        self._is_unknown_title(right.song_title)):
        bscore_left = self._lookup_boundary_score(left.end_time)
        bscore_right = self._lookup_boundary_score(right.start_time)
        if gap < 15.0 and max(bscore_left, bscore_right) < 0.70:
            return True, f"both_weak_wide_gap({gap:.0f}s)"
    
    # ===== 新增条件 (e)(f) =====
    # (e) 音频特征相似性：能量指纹相关
    left_fp = getattr(left, '_energy_fingerprint', None)
    right_fp = getattr(right, '_energy_fingerprint', None)
    if left_fp is not None and right_fp is not None:
        from scipy.spatial.distance import correlation
        corr = 1.0 - correlation(left_fp, right_fp)
        if corr > 0.85 and gap < 20.0:
            return True, f"energy_corr({corr:.2f})gap({gap:.0f}s)"
    
    # (f) 时长过短：任一歌曲 < 30s 且间隔小
    min_dur = min(left.end_time - left.start_time,
                   right.end_time - right.start_time)
    if min_dur < 30 and gap < 10.0:
        return True, f"too_short({min_dur:.0f}s)"
    
    return False, ""
```

#### 优化 C1：注入 AST 场景纠偏（填补重大遗漏）

具体改动 (`processor.py` 在 `_refine_songformer_segments_labels` 之后调用):

```python
def _inject_ast_scene_correction(self, segments, y_song, sr, song_start):
    """
    C1: 用 AST/Audio Features 做 scene 纠偏
    高置信度的 speech/talk/crowd 直接覆盖 SongFormer 结果
    """
    try:
        from src.audio_analyzer import SegmentClassifier
        classifier = SegmentClassifier()
        
        for seg in segments:
            local_start = seg.start_time - song_start
            local_end = seg.end_time - song_start
            
            if local_end <= local_start or local_end - local_start < 5:
                continue
            
            y_seg = y_song[int(local_start * sr):int(local_end * sr)]
            if len(y_seg) < sr * 3:
                continue
            
            features = self._extract_segment_features(
                y_seg, sr, sr / 1024.0,
                global_stats={'rms_mean': 0.05}, song_stats={}
            )
            
            context = {
                'position_ratio': local_start / max((seg.end_time - seg.start_time), 1e-8),
                'relative_energy': features.relative_energy,
            }
            scores = classifier._score_all(features, context)
            
            speech_score = scores.get(LABEL_SPEECH, 0)
            talk_score = scores.get(LABEL_TALK, 0)
            crowd_score = scores.get(LABEL_CROWD, 0)
            chorus_score = scores.get(LABEL_CHORUS, 0)
            
            # 规则1: speech/talk 高置信 -> 强制覆盖
            if speech_score >= 6.0 or talk_score >= 6.0:
                old_label = seg.label
                seg.label = LABEL_SPEECH
                seg.label_source = 'ast_override'
                logger.info(f"[AST纠偏] {old_label}→speech @ {seg.start_time:.0f}s "
                            f"(sp={speech_score:.1f},talk={talk_score:.1f})")
            
            # 规则2: crowd/audience 中高置信 -> 覆盖 chorus
            elif (crowd_score >= 5.5 or audience_score >= 5.5):
                if seg.label == LABEL_CHORUS:
                    seg.label = LABEL_AUDIENCE
                    seg.label_source = 'ast_override'
                elif seg.label == LABEL_VERSE:
                    seg.label = LABEL_CROWD
                    seg.label_source = 'ast_override'
            
            # 规则3: music 高置信 + chorus 低置信 -> 保底为 verse
            elif chorus_score < 4.0 and seg.label == LABEL_CHORUS:
                rel_e = getattr(features, 'relative_energy', 1.0)
                if rel_e < 1.05:
                    seg.label = LABEL_VERSE
                    seg.label_source = 'ast_downgrade'
                    
    except Exception as e:
        logger.warning(f"[AST纠偏] 失败（非阻塞）: {e}")
    
    return segments
```

注意: 这里复用了 audio_analyzer.py 中已有的 SegmentClassifier._score_all() 方法，关键是之前从未在SongFormer路径中调用它。

#### 优化 C2：新增时序平滑（Viterbi 风格 DP）

目标: 消除短时标签抖动（如 chorus->verse->chorus->verse 快速切换）

```python
def _temporal_smooth(self, segments, min_duration=6.0, max_duration=None):
    """C2: 时序平滑 — Viterbi 思想的最短路径平滑"""
    if len(segments) <= 2:
        return segments
    
    result = [segments[0].copy()]
    
    # 正向扫描: 合并短抖动段
    for i in range(1, len(segments)):
        prev = result[-1]
        curr = segments[i]
        dur = curr.end_time - curr.start_time
        
        # 条件1: 当前段太短 且 与前一段同标签 -> 合并
        if dur < min_duration and curr.label == prev.label:
            prev.end_time = curr.end_time
            prev.confidence = (prev.confidence * 0.7 + curr.confidence * 0.3)
        else:
            if dur < min_duration * 0.5:
                if i < len(segments) - 1:
                    next_seg = segments[i + 1]
                    if next_seg.label == prev.label:
                        prev.end_time = curr.end_time
                        continue
            result.append(curr.copy())
    
    # 反向扫描: 处理尾部短段
    smoothed = [result[0]]
    for i in range(1, len(result)):
        curr = result[i]
        prev = smoothed[-1]
        dur = curr.end_time - curr.start_time
        
        if dur < min_duration and curr.label == prev.label:
            prev.end_time = curr.end_time
            prev.confidence = (prev.confidence * 0.7 + curr.confidence * 0.3)
        else:
            smoothed.append(curr.copy())
    
    return smoothed
```

参数建议: min_duration=6.0（快歌降到4.0，慢歌提到8.0）

#### 优化 C3：副歌过预测抑制增强

目标: 解决"全是副歌"

```python
def _enhanced_chorus_suppression(self, export_segments, song):
    """C3: 增强版副歌抑制 基于"正常歌曲通常只有 2-4 个副歌段"先验"""
    chorus_count = sum(1 for s in export_segments if s['type'] == '副歌')
    total = len(export_segments)
    
    if total < 3 or chorus_count <= 0:
        return export_segments
    
    chorus_ratio = chorus_count / total
    
    # 规则1: 副歌占比 > 65% 触发抑制
    if chorus_ratio > 0.65:
        sorted_segs = sorted(
            [s for s in export_segments if s['type'] == '副歌'],
            key=lambda x: -(x.get('segment', {}).get('confidence', 0) or 0)
        )
        n_demote = int((chorus_count - total * 0.45))
        n_demote = max(0, min(n_demote, len(sorted_segs)))
        demoted = set(id(s) for s in sorted_segs[:n_demote])
        for s in export_segments:
            if s['type'] == '副歌' and id(s.get('segment', {})) in demoted:
                s['type'] = '主歌'; s['_demoted'] = True
    
    # 规则2: 连续超过3个chorus 中间每隔一个降级
    consecutive = 0
    for i, s in enumerate(export_segments):
        if s['type'] == '副歌':
            consecutive += 1
            if consecutive >= 3 and (i - consecutive) >= 1:
                if consecutive % 2 == 1:
                    s['type'] = '导歌'; s['_demoted'] = True
                    consecutive = 0
        else:
            consecutive = 0
    
    return export_segments
```
